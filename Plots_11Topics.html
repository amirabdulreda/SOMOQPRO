day_dict = {}
for index,row in RMS_csv.iterrows():
    day = row['date']
    topic_id = row['nmf_w2v_topic_id']
    user_id = row['User_ID']
    
    topic_dict = day_dict.get(day, {})
    user_lst = topic_dict.get(topic_id,[])
    user_lst.append(user_id)
    
    topic_dict[topic_id] = user_lst
    day_dict[day] = topic_dict

days = sorted(list(set(RMS_csv["date"].values)))
topics = sorted(list(set(RMS_csv["nmf_w2v_topic_id"].values)))

topic_daily_unique = {}
topic_daily = {}
for day in days:
    for topic in topics:
        tlst = topic_daily.get(topic,[])
        tlst_unique= topic_daily_unique.get(topic,[])
        
        if topic in day_dict[day]:
            tlst.append(len(day_dict[day][topic]))
            tlst_unique.append(len(set(day_dict[day][topic])))
        else:
            tlst.append(0)
            tlst_unique.append(0)
            
        topic_daily_unique[topic] = tlst_unique
        topic_daily[topic] = tlst
